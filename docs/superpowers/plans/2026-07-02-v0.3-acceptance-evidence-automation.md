# v0.3 Acceptance Evidence Automation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade Devflow verification from manually entered acceptance evidence to a unified evidence system that can record manual, HTTP/API, link/report, monitor, and log evidence, with automatic fetch adapters and deterministic pass/fail/blocked evaluation.

**Architecture:** Keep `internal/demandflow` as the owner of demand state and artifact/event writes. Add a focused `internal/evidence` package for redaction and external evidence fetch adapters that return safe, normalized evidence payloads. CLI commands orchestrate adapter calls and then persist evidence through `demandflow.AddEvidence`, so every evidence path writes to both `verification.md` and `events.jsonl`.

**Tech Stack:** Go standard library only (`net/http`, `httptest`, `regexp`, `time`, `io`), existing `internal/artifacts`, existing `internal/demandflow`, existing CLI pattern, PowerShell release-readiness fake server smoke. No new dependencies.

## Evidence Semantics

Unified evidence statuses:

```text
pass     evidence matched expectation
fail     request/link succeeded, but evidence did not match expectation
blocked  network error, timeout, permission problem, unsupported method, or inaccessible link
```

Evidence write contract:

- Every successful `evidence add` or `evidence fetch` appends to `verification.md`.
- Every successful `evidence add` or `evidence fetch` appends `verification.evidence_recorded` to `events.jsonl`.
- `verification.md` section heading must be `## Acceptance Evidence`.
- Existing old heading `## Manual Acceptance Evidence` does not need migration for old artifacts, but new writes must use the new heading.
- Event type stays `verification.evidence_recorded` for compatibility.
- Event data should include at least:

```text
type
criterion
status
summary
link
by
source
method
url
expected_status
actual_status
expect_contains
evidence_file
```

Only include fields that apply. Values must be redacted.

## Secret Redaction Rules

Redaction must run before writing artifacts/events.

Redact these keys in headers, URLs, query strings, JSON-like text, and summaries:

```text
authorization
cookie
set-cookie
token
access_token
refresh_token
api_key
apikey
secret
password
passwd
session
signature
client_secret
tenant_access_token
```

Examples:

```text
Authorization: Bearer abc123          -> Authorization: Bearer ***
Cookie: sid=abc; theme=dark           -> Cookie: ***
https://x.test?a=1&token=abc          -> https://x.test?a=1&token=***
{"api_key":"abc","ok":true}           -> {"api_key":"***","ok":true}
```

Body recording limits:

- Store only a short response excerpt, default max 4096 bytes.
- The excerpt must be redacted.
- If response is larger, append `... (truncated)`.

```go
package evidence

import (
	"regexp"
	"strings"
	"unicode/utf8"
)

type redactionRule struct {
	pattern     *regexp.Regexp
	replacement string
}

var redactionRules = []redactionRule{
	{regexp.MustCompile(`(?i)(authorization:\s*bearer\s+)[^\s]+`), `${1}***`},
	{regexp.MustCompile(`(?i)(authorization:\s*basic\s+)[^\s]+`), `${1}***`},
	{regexp.MustCompile(`(?i)(cookie:\s*).+`), `${1}***`},
	{regexp.MustCompile(`(?i)(set-cookie:\s*).+`), `${1}***`},
	{regexp.MustCompile(`(?i)([?&](?:token|access_token|refresh_token|api_key|apikey|secret|password|passwd|session|signature|client_secret|tenant_access_token)=)[^&\s]+`), `${1}***`},
	{regexp.MustCompile(`(?i)("(?:token|access_token|refresh_token|api_key|apikey|secret|password|passwd|session|signature|client_secret|tenant_access_token)"\s*:\s*")[^"]*(")`), `${1}***${2}`},
}

func Redact(value string) string {
	out := value
	for _, rule := range redactionRules {
		out = rule.pattern.ReplaceAllString(out, rule.replacement)
	}
	return out
}

func Excerpt(value string, maxBytes int) string {
	if maxBytes <= 0 {
		maxBytes = 4096
	}
	redacted := Redact(value)
	if len(redacted) <= maxBytes {
		return redacted
	}
	truncated := redacted[:maxBytes]
	for !utf8.ValidString(truncated) && len(truncated) > 0 {
		truncated = truncated[:len(truncated)-1]
	}
	return truncated + "... (truncated)"
}
```

Create `internal/evidence/http.go`:

```go
package evidence

import (
	"bytes"
	"context"
	"fmt"
	"io"
	"net/http"
	"strings"
	"time"
)

type HTTPFetchRequest struct {
	URL            string
	Method         string
	Body           string
	Headers        map[string]string
	ExpectStatus   int
	ExpectContains string
	Timeout        time.Duration
}

type FetchResult struct {
	Status       string
	Summary      string
	URL          string
	Method       string
	ActualStatus int
}

type HTTPFetcher struct {
	Client *http.Client
}

func (f HTTPFetcher) Fetch(ctx context.Context, input HTTPFetchRequest) FetchResult {
	method := strings.ToUpper(strings.TrimSpace(input.Method))
	if method == "" {
		method = http.MethodGet
	}
	timeout := input.Timeout
	if timeout <= 0 {
		timeout = 10 * time.Second
	}
	ctx, cancel := context.WithTimeout(ctx, timeout)
	defer cancel()

	req, err := http.NewRequestWithContext(ctx, method, strings.TrimSpace(input.URL), bytes.NewBufferString(input.Body))
	if err != nil {
		return FetchResult{Status: "blocked", URL: Redact(input.URL), Method: method, Summary: "build request failed: " + Redact(err.Error())}
	}
	for name, value := range input.Headers {
		req.Header.Set(name, value)
	}
	client := f.Client
	if client == nil {
		client = http.DefaultClient
	}
	resp, err := client.Do(req)
	if err != nil {
		return FetchResult{Status: "blocked", URL: Redact(input.URL), Method: method, Summary: "request failed: " + Redact(err.Error())}
	}
	defer resp.Body.Close()
	data, _ := io.ReadAll(io.LimitReader(resp.Body, 4096+1))
	excerpt := Excerpt(string(data), 4096)

	statusOK := input.ExpectStatus == 0 || resp.StatusCode == input.ExpectStatus
	bodyOK := strings.TrimSpace(input.ExpectContains) == "" || strings.Contains(string(data), input.ExpectContains)
	status := "pass"
	if !statusOK || !bodyOK {
		status = "fail"
	}
	summary := fmt.Sprintf("%s %s returned %d", method, Redact(input.URL), resp.StatusCode)
	if input.ExpectStatus != 0 {
		summary += fmt.Sprintf(" expected_status=%d", input.ExpectStatus)
	}
	if input.ExpectContains != "" {
```

```go
package evidence

import (
	"context"
	"fmt"
	"net/http"
	"strings"
	"time"
)

type LinkFetchRequest struct {
	URL          string
	ExpectStatus int
	Timeout      time.Duration
}

type LinkFetcher struct {
	Client *http.Client
}

func (f LinkFetcher) Fetch(ctx context.Context, input LinkFetchRequest) FetchResult {
	result, retryWithGet := f.fetchWithMethod(ctx, http.MethodHead, input)
	if retryWithGet {
		result, _ = f.fetchWithMethod(ctx, http.MethodGet, input)
	}
	return result
}

func (f LinkFetcher) fetchWithMethod(ctx context.Context, method string, input LinkFetchRequest) (FetchResult, bool) {
	timeout := input.Timeout
	if timeout <= 0 {
		timeout = 10 * time.Second
	}
	ctx, cancel := context.WithTimeout(ctx, timeout)
	defer cancel()
	req, err := http.NewRequestWithContext(ctx, method, strings.TrimSpace(input.URL), nil)
	if err != nil {
		return FetchResult{Status: "blocked", URL: Redact(input.URL), Method: method, Summary: "build link request failed: " + Redact(err.Error())}, false
	}
	client := f.Client
	if client == nil {
		client = http.DefaultClient
	}
	resp, err := client.Do(req)
	if err != nil {
		return FetchResult{Status: "blocked", URL: Redact(input.URL), Method: method, Summary: "link request failed: " + Redact(err.Error())}, false
	}
	defer resp.Body.Close()
	if method == http.MethodHead && resp.StatusCode == http.StatusMethodNotAllowed {
		return FetchResult{}, true
	}
	expected := input.ExpectStatus
	if expected == 0 {
		expected = http.StatusOK
	}
	status := "pass"
	if resp.StatusCode != expected {
		status = "fail"
	}
	return FetchResult{Status: status, URL: Redact(input.URL), Method: method, ActualStatus: resp.StatusCode, Summary: fmt.Sprintf("%s %s returned %d expected_status=%d", method, Redact(input.URL), resp.StatusCode, expected)}, false
}
```
