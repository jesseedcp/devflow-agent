# v0.4.1 Codemap Hardening And v0.5 Plan Grounding Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Harden the v0.4 codemap release, then turn codemap results into a concrete plan-grounding and change-scope loop so Devflow can tell whether an implementation plan is tied to the right code areas.

**Architecture:** v0.4.1 is a small hardening release that keeps codemap Go-only and deterministic while improving stale-index behavior, scan exclusions, and operator guidance. v0.5 adds demand-level planning artifacts (`plan-context.md` and `change-scope.md`) built from requirements, memory context, and codemap facts; evaluation then checks whether plans and implementation diffs remain grounded in that declared scope.

**Tech Stack:** Go standard library only (`go/parser`, `go/ast`, `encoding/json`, `filepath`, `strings`, `os/exec`), existing `internal/codemap`, `internal/artifacts`, `internal/demandflow`, existing CLI pattern, PowerShell release-readiness smokes. No embeddings, no deep call graph, no wiki write-back, no automatic code editing.

```go
package cli

import (
	"flag"
	"fmt"
	"io"
	"os"
	"path/filepath"
	"strings"

	"github.com/jesseedcp/devflow-agent/internal/artifacts"
)

func runPlanContext(args []string, stdout io.Writer, stderr io.Writer) error {
	if len(args) == 0 {
		return fmt.Errorf("plan-context requires a subcommand: refresh")
	}
	switch args[0] {
	case "refresh":
		return runPlanContextRefresh(args[1:], stdout, stderr)
	default:
		return fmt.Errorf("unknown plan-context command %q", args[0])
	}
}

func runPlanContextRefresh(args []string, stdout io.Writer, stderr io.Writer) error {
	fs := flag.NewFlagSet("plan-context refresh", flag.ContinueOnError)
	fs.SetOutput(stderr)
	var root, demandID string
	fs.StringVar(&root, "root", ".", "root directory")
	fs.StringVar(&demandID, "demand", "", "demand id")
	if err := fs.Parse(args); err != nil {
		return err
	}
	root = normalizedRoot(root)
	demandID = strings.TrimSpace(demandID)
	if demandID == "" {
		return fmt.Errorf("--demand is required")
	}
	store := artifacts.NewStore(root)
	demand, err := store.LoadDemand(demandID)
	if err != nil {
		return err
	}
	demandDir := store.DemandDir(demandID)
	requirements := readOptionalArtifact(demandDir, artifacts.RequirementsFile)
	context := readOptionalArtifact(demandDir, artifacts.ContextFile)
	codemap := readOptionalArtifact(demandDir, artifacts.CodemapFile)
	report := renderPlanContext(demand.Title, requirements, context, codemap)
	if err := store.WriteArtifact(demandID, artifacts.PlanContextFile, report); err != nil {
		return err
	}
	_, err = fmt.Fprintf(stdout, "plan context refreshed for %s\n", demandID)
	return err
}

func readOptionalArtifact(demandDir, name string) string {
	data, err := os.ReadFile(filepath.Join(demandDir, name))
	if err != nil {
		return ""
	}
	return string(data)
}

func renderPlanContext(title, requirements, context, codemap string) string {
	var builder strings.Builder
	fmt.Fprintf(&builder, "# Plan Context: %s\n\n", title)
	builder.WriteString("## Requirements\n\n")
	builder.WriteString(strings.TrimSpace(requirements))
	builder.WriteString("\n\n## Memory Context\n\n")
	builder.WriteString(strings.TrimSpace(context))
	builder.WriteString("\n\n## Codemap Context\n\n")
	builder.WriteString(strings.TrimSpace(codemap))
	builder.WriteString("\n")
	return builder.String()
}
```

```go
package scope

import (
	"fmt"
	"strings"
)

func RenderDeclaration(title string, decl Declaration) string {
	var b strings.Builder
	fmt.Fprintf(&b, "# Change Scope: %s\n\n", title)
	b.WriteString("## Source Files\n\n")
	writeBullets(&b, decl.SourceFiles)
	b.WriteString("\n## Test Files\n\n")
	writeBullets(&b, decl.TestFiles)
	b.WriteString("\n## Out Of Scope\n\n")
	writeBullets(&b, decl.OutOfScope)
	return b.String()
}

func ParseDeclaration(text string) Declaration {
	var decl Declaration
	section := ""
	for _, line := range strings.Split(strings.ReplaceAll(text, "\r\n", "\n"), "\n") {
		trimmed := strings.TrimSpace(line)
		lower := strings.ToLower(trimmed)
		if strings.HasPrefix(lower, "## source files") {
			section = "source"
			continue
		}
		if strings.HasPrefix(lower, "## test files") {
			section = "test"
			continue
		}
		if strings.HasPrefix(lower, "## out of scope") {
			section = "out"
			continue
		}
		if !strings.HasPrefix(trimmed, "- ") {
			continue
		}
		value := strings.TrimSpace(strings.TrimPrefix(trimmed, "- "))
		value = strings.Trim(value, "` ")
		if value == "" {
			continue
		}
		switch section {
		case "source":
			decl.SourceFiles = append(decl.SourceFiles, value)
		case "test":
			decl.TestFiles = append(decl.TestFiles, value)
		case "out":
			decl.OutOfScope = append(decl.OutOfScope, value)
		}
	}
	return decl
}

func writeBullets(b *strings.Builder, values []string) {
	if len(values) == 0 {
		b.WriteString("- None declared\n")
		return
	}
	for _, value := range values {
		value = strings.TrimSpace(value)
		if value == "" {
			continue
		}
		fmt.Fprintf(b, "- `%s`\n", value)
	}
}
```

```go
package scope

type DiffResult struct {
	InScope      []string
	OutOfScope   []string
	MissingTests  []string
	DeclaredFiles []string
}

func CompareChangedFiles(decl Declaration, changed []string) DiffResult {
	declared := map[string]bool{}
	testDeclared := map[string]bool{}
	var declaredFiles []string
	for _, file := range append([]string{}, decl.SourceFiles...) {
		declared[file] = true
		declaredFiles = append(declaredFiles, file)
	}
	for _, file := range decl.TestFiles {
		declared[file] = true
		testDeclared[file] = true
		declaredFiles = append(declaredFiles, file)
	}
	changedSet := map[string]bool{}
	var result DiffResult
	result.DeclaredFiles = declaredFiles
	for _, file := range changed {
		changedSet[file] = true
		if declared[file] {
			result.InScope = append(result.InScope, file)
		} else {
			result.OutOfScope = append(result.OutOfScope, file)
		}
	}
	for file := range testDeclared {
		if !changedSet[file] {
			result.MissingTests = append(result.MissingTests, file)
		}
	}
	return result
}
```
